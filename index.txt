    <script>
        let apostas = JSON.parse(localStorage.getItem('apostasBetania')) || [];
        let grafico;

        function iniciarGrafico() {
            const ctx = document.getElementById('graficoResultados').getContext('2d');
            grafico = new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels: ['Green', 'Red'],
                    datasets: [{
                        data: [0, 0],
                        backgroundColor: ['#34C759', '#FF3B30'],
                        borderWidth: 0
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: {
                            position: 'bottom',
                            labels: { font: { size: 14 } }
                        }
                    }
                }
            });
        }

        function mudarPagina(pagina) {
            document.getElementById('paginaHome').classList.toggle('hidden', pagina !== 'home');
            document.getElementById('paginaHistorico').classList.toggle('hidden', pagina !== 'historico');
            
            if (pagina === 'home') {
                document.getElementById('tabHome').classList.add('tab-active');
                document.getElementById('tabHome').classList.remove('tab-inactive');
                document.getElementById('tabHistorico').classList.add('tab-inactive');
                document.getElementById('tabHistorico').classList.remove('tab-active');
            } else {
                document.getElementById('tabHistorico').classList.add('tab-active');
                document.getElementById('tabHistorico').classList.remove('tab-inactive');
                document.getElementById('tabHome').classList.add('tab-inactive');
                document.getElementById('tabHome').classList.remove('tab-active');
            }
        }

        function selecionarResultado(tipo) {
            document.getElementById('resultado').value = tipo;
            document.getElementById('btnGreen').classList.toggle('active', tipo === 'green');
            document.getElementById('btnRed').classList.toggle('active', tipo === 'red');
        }

        function abrirModal() {
            document.getElementById('modalAposta').classList.remove('hidden');
            document.body.style.overflow = 'hidden';
            document.getElementById('formAposta').reset();
            selecionarResultado('');
        }

        function fecharModal() {
            document.getElementById('modalAposta').classList.add('hidden');
            document.body.style.overflow = 'auto';
        }

        function salvarAposta(e) {
            e.preventDefault();
            const valor = parseFloat(document.getElementById('valorApostado').value);
            const odd = parseFloat(document.getElementById('odd').value);
            const resultado = document.getElementById('resultado').value;
            const obs = document.getElementById('observacao').value;
            const data = new Date();

            if (!resultado) {
                alert('Selecione o resultado: Green ou Red!');
                return;
            }

            const retornoTotal = valor * odd;
            const lucro = resultado === 'green' ? retornoTotal - valor : -valor;

            const nova = {
                id: Date.now(),
                valor,
                odd,
                retorno: retornoTotal,
                lucro,
                resultado,
                obs,
                data: data.toLocaleDateString('pt-BR'),
                timestamp: data.getTime()
            };

            apostas.unshift(nova);
            localStorage.setItem('apostasBetania', JSON.stringify(apostas));
            fecharModal();
            atualizarDados();
        }

        function atualizarDados() {
            let totalInvestido = 0;
            let totalLucro = 0;
            let qtdGreen = 0, qtdRed = 0;
            let balancoDia = 0;
            const hoje = new Date().toLocaleDateString('pt-BR');
            const dataInicial = apostas.length ? new Date(apostas[apostas.length - 1].timestamp) : new Date();

            apostas.forEach(a => {
                totalInvestido += a.valor;
                totalLucro += a.lucro;
                if (a.resultado === 'green') qtdGreen++;
                else qtdRed++;
                if (a.data === hoje) balancoDia += a.lucro;
            });

            const rentabilidadeTotal = totalInvestido > 0 ? ((totalLucro / totalInvestido) * 100).toFixed(2) : 0;
            const diferencaDias = apostas.length ? Math.ceil((new Date() - dataInicial) / (1000 * 60 * 60 * 24)) : 0;
            const tempoTexto = diferencaDias < 30 ? `${diferencaDias} dias` : diferencaDias < 60 ? `1 mês` : `${Math.floor(diferencaDias / 30)} meses`;

            document.getElementById('saldoTotal').textContent = totalLucro.toFixed(2).replace('.', ',');
            document.getElementById('balancoDia').textContent = balancoDia.toFixed(2).replace('.', ',');
            document.getElementById('rentabilidadeDia').textContent = balancoDia !== 0 ? `${((balancoDia / (totalInvestido || 1)) * 100).toFixed(2)}%` : '--';
            document.getElementById('valorInvestido').textContent = totalInvestido.toFixed(2).replace('.', ',');
            document.getElementById('balancoTotal').textContent = `R$ ${totalLucro.toFixed(2).replace('.', ',')}`;
            document.getElementById('balancoTotal').className = `text-xl font-semibold ${totalLucro >= 0 ? 'text-green' : 'text-red'}`;
            document.getElementById('rentabilidadeTotal').textContent = `${rentabilidadeTotal}%`;
            document.getElementById('tempoTotal').textContent = tempoTexto;
            document.getElementById('totalApostas').textContent = apostas.length;

            grafico.data.datasets[0].data = [qtdGreen, qtdRed];
            grafico.update();

            atualizarHistorico();
        }

        function atualizarHistorico() {
            const lista = document.getElementById('listaHistorico');
            if (apostas.length === 0) {
                lista.innerHTML = `<div class="text-center text-gray-400 py-10"><i class="fa fa-history text-4xl mb-3"></i><p>Nenhuma aposta registrada ainda</p></div>`;
                return;
            }
            lista.innerHTML = '';
            apostas.forEach(a => {
                const cor = a.resultado === 'green' ? 'text-green' : 'text-red';
                const icone = a.resultado === 'green' ? 'fa-arrow-up' : 'fa-arrow-down';
                lista.innerHTML += `
                    <div class="bg-white rounded-xl p-4 shadow-sm flex items-center justify-between">
                        <div class="flex items-center gap-3">
                            <div class="w-10 h-10 rounded-full flex items-center justify-center ${cor} bg-opacity-10 bg-${cor.replace('text-', '')}/10">
                                <i class="fa ${icone}"></i>
                            </div>
                            <div>
                                <p class="font-medium">${a.obs || 'Aposta'}</p>
                                <p class="text-xs text-gray-500">${a.data} • Apostado: R$ ${a.valor.toFixed(2).replace('.', ',')} • Odd: ${a.odd.toFixed(2)}</p>
                            </div>
                        </div>
                        <div class="text-right">
                            <p class="font-semibold ${cor}">R$ ${a.lucro.toFixed(2).replace('.', ',')}</p>
                            <p class="text-xs text-gray-500">${a.resultado.toUpperCase()}</p>
                        </div>
                    </div>`;
            });
        }

        window.onload = () => {
            iniciarGrafico();
            atualizarDados();
        };
    </script>
